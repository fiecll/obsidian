-ユークリッド距離を返す関数として、hypot(x,y)が存在する

---
#include<stdio.h>
#include<math.h>

int main(){
	int n;
	scanf("%d", &n);

	double ans = 0;
	int crrx = 0;
	int crry = 0;
	for(int i=0; i<n; i++){
		int x, y;
		scanf("%d%d", &x, &y);
		ans += hypot(x-crrx, y-crry);
		crrx = x;
		crry = y;
	}
	ans += hypot(crrx, crry);
	printf("%.10f\n", ans);
}
